/* Variables Core - Sistema Sergio Forés */
:root {
  /* Tipografía fluida Utopia */
  --step-0: clamp(1rem, 0.93rem + 0.37vw, 1.25rem);
  --step-1: clamp(1.2rem, 1.11rem + 0.54vw, 1.56rem);
  --step-2: clamp(1.44rem, 1.32rem + 0.78vw, 1.95rem);
  
  /* Espaciado matemático */
  --space-s: clamp(1rem, 0.93rem + 0.37vw, 1.25rem);
  --space-m: clamp(1.5rem, 1.39rem + 0.56vw, 1.88rem);
  --space-l: clamp(2rem, 1.85rem + 0.74vw, 2.5rem);
  
  /* Paleta Sergio Forés */
  --color-yellow: #FFFF00;
  --color-cyan: #00FFFF;
  --color-red: #FF0000;
  --color-black: #000000;
  --color-white: #FFFFFF;

  --gray-1: #333333;
  --gray-2: #666666;
  --gray-3: #999999;
  --gray-4: #CCCCCC;
  --gray-5: #EEEEEE;

  --color-primary: var(--color-black);
  --color-accent: var(--color-yellow);
  --color-background: var(--color-white);
  --color-surface: var(--gray-5);
}

  padding: var(--space-m);
  background: var(--color-background);
}

}

  padding: var(--space-s) var(--space-m);

  color: var(--color-white);
